## Quick Quick Start (The "I'm in a Hurry" Version) âš¡

Need CLS working in 30 seconds? Here you go:

```typescript
import { cls } from '@use-pico/cls';

// Define a simple button
const ButtonCls = cls({
  tokens: {
    "color.bg": ["default", "primary"]
  },
  slot: ["root"],
  variant: {
    tone: ["default", "primary"]
  }
}, ({ what, def }) => ({
  token: def.token({
    "color.bg": {
      default: ["bg-gray-100"],
      primary: ["bg-blue-600"]
    }
  }),
  rules: [
    def.root({
      root: what.both(["px-4", "py-2", "rounded"], ["color.bg.default"])
    }),
    def.rule(
      what.variant({ tone: "primary" }),
      { root: what.token(["color.bg.primary"]) }
    )
  ],
  defaults: def.defaults({
    tone: "default"
  })
}));

// Use it!
const classes = ButtonCls.create(({ what }) => ({
  variant: what.variant({ tone: "primary" })
}));

// In your component
<button className={classes.root()}>Click me!</button>
```

**Boom!** You just built a type-safe, token-based button system in 18 lines! ðŸŽ¯

## Less Quick Start (The "Show Me the Real Power" Version) ðŸš€

Ready to see what CLS can really do? Let's build a proper design system with tokens and inheritance!

### Step 1: Create Your Theme (Tokens Only) ðŸŽ¨

```typescript
import { cls } from '@use-pico/cls';

// Start with just tokens - your design system foundation
const ThemeCls = cls({
  tokens: {
    "color.text": ["default", "primary", "secondary", "danger"],
    "color.bg": ["default", "primary", "secondary", "danger"],
    "spacing.padding": ["sm", "md", "lg"],
    "border.radius": ["sm", "md", "lg"]
  },
  slot: [], // No slots for theme-only
  variant: {} // No variants for theme-only
}, ({ what, def }) => ({
  token: def.token({
    "color.text": {
      default: ["text-gray-900"],
      primary: ["text-blue-600"],
      secondary: ["text-gray-600"],
      danger: ["text-red-600"]
    },
    "color.bg": {
      default: ["bg-gray-100"],
      primary: ["bg-blue-600"],
      secondary: ["bg-gray-500"],
      danger: ["bg-red-500"]
    },
    "spacing.padding": {
      sm: ["px-2", "py-1"],
      md: ["px-4", "py-2"],
      lg: ["px-6", "py-3"]
    },
    "border.radius": {
      sm: ["rounded"],
      md: ["rounded-md"],
      lg: ["rounded-lg"]
    }
  }),
  rules: [], // No rules for theme-only
  defaults: def.defaults({}) // No defaults for theme-only
}));
```

### Step 2: Build Your Button (Using Theme Tokens) ðŸŽ¯

```typescript
// Extend the theme to create a button module
const ButtonCls = ThemeCls.extend({
  tokens: {
    "color.text": ["default", "primary", "secondary", "danger"],
    "color.bg": ["default", "primary", "secondary", "danger"]
  },
  slot: ["root", "label", "icon"],
  variant: {
    size: ["sm", "md", "lg"],
    tone: ["default", "primary", "secondary", "danger"],
    disabled: ["bool"]
  }
}, ({ what, def }) => ({
  token: def.token({
    "color.text": {
      default: ["text-gray-900"],
      primary: ["text-white"],
      secondary: ["text-white"],
      danger: ["text-white"]
    },
    "color.bg": {
      default: ["bg-gray-100"],
      primary: ["bg-blue-600"],
      secondary: ["bg-gray-500"],
      danger: ["bg-red-500"]
    }
  }),
  rules: [
    def.root({
      root: what.both(
        ["inline-flex", "items-center", "justify-center", "font-medium", "transition-colors"],
        ["color.text.default", "color.bg.default", "spacing.padding.md", "border.radius.md"]
      ),
      label: what.css(["font-medium"]),
      icon: what.css(["mr-2"])
    }),
    def.rule(
      what.variant({ tone: "primary" }),
      { root: what.token(["color.text.primary", "color.bg.primary"]) }
    ),
    def.rule(
      what.variant({ tone: "secondary" }),
      { root: what.token(["color.text.secondary", "color.bg.secondary"]) }
    ),
    def.rule(
      what.variant({ tone: "danger" }),
      { root: what.token(["color.text.danger", "color.bg.danger"]) }
    ),
    def.rule(
      what.variant({ size: "sm" }),
      { root: what.token(["spacing.padding.sm", "border.radius.sm"]) }
    ),
    def.rule(
      what.variant({ size: "lg" }),
      { root: what.token(["spacing.padding.lg", "border.radius.lg"]) }
    ),
    def.rule(
      what.variant({ disabled: true }),
      {
        root: what.css(["opacity-50", "cursor-not-allowed"]),
        label: what.css(["text-gray-500"])
      }
    )
  ],
  defaults: def.defaults({
    size: "md",
    tone: "default",
    disabled: false
  })
}));
```

## What Just Happened? ðŸ¤¯

1. **ThemeCls**: Pure tokens, no slots or variants - just your design system foundation
2. **ButtonCls**: Extends ThemeCls, adds slots and variants, uses theme tokens
3. **Type Safety**: TypeScript knows exactly what variants and tokens are available
4. **Consistency**: All buttons use the same tokens from your theme
5. **Flexibility**: Easy to override at runtime if needed

## Pro Tips ðŸ’¡

### Runtime Overrides
```typescript
// Override tokens for a specific use case
const customButton = ButtonCls.create(({ what, override }) => ({
  variant: what.variant({ tone: "primary" }),
  token: override.token({
    "color.bg": {
      primary: ["bg-purple-600"] // Override the blue with purple
    }
  })
}));

// Use it
<button className={customButton.root()}>Purple Button</button>
```

### Slot Overrides
```typescript
// Add extra classes to specific slots
const buttonWithShadow = ButtonCls.create(({ what }) => ({
  variant: what.variant({ tone: "primary" }),
  slot: {
    root: what.css(["shadow-lg", "hover:shadow-xl"])
  }
}));

// Use it
<button className={buttonWithShadow.root()}>Shadow Button</button>
```
